<?php
// Parent class
abstract class A
 {
  public $name;
  public function __construct($name)
 {
    $this->name = $name;
  }
  abstract public function printdata();
}

// Child classes
class B extends A
 {
  public function printdata()
 {
    return "We are in class: $this->name!";
  }
}

class C extends A {
  public function printdata()
 {
    return "We are in class: $this->name!";
  }
}



// Create objects from the child classes
$b = new B("class B");
echo $b->printdata();
echo "<br>";

$c = new C("class C");
echo $c->printdata();
echo "<br>";


?>
